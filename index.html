Gatling simulation setup

  • nothingFor(4) // Verilen süre için bekler
  • atOnceUsers(10) // Anlık olarak verilen kullanıcı kadar ekler
  • rampUsers(10) over(5 seconds) // Verilen süre içinde belirtilen kadar kullanıcı doğrusal olarak ekler
  • constantUsersPerSec(20) during(15 seconds) // Verilen süre içinde düzenli aralıklarla saniye başına kullanıcı olarak ekler
  • constantUsersPerSec(20) during(15 seconds) randomized // Verilen süre içinde düzensiz aralıklarla saniye başına kullanıcı olarak ekler
  • rampUsersPerSec(10) to(20) during(10 minutes) // Verilen aralıktaki kullanıcıları belirtilen süre içinde düzenli olarak ekler
  • rampUsersPerSec(10) to(20) during(10 minutes) randomized // Verilen aralıktaki kullanıcıları belirtilen süre içinde düzensiz olarak ekler
  • splitUsers(1000) into(rampUsers(10) over(10 seconds)) separatedBy(10 seconds) // Verilen kullanıcı sayısına ulaşana kadar belirtilen injection adımını istenen süre aralığında tekrarlar
  • splitUsers(1000) into(rampUsers(10) over(10 seconds)) separatedBy(atOnceUsers(30)) // Verilen kullanıcı sayısına ulaşana kadar belirtilen injection adımını, belirtilen ikinci injection adımı süre aralığında tekrarlar
